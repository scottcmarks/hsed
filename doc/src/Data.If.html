module Data.If ((?)) where

import           GHC.Types (Bool)

-- | @if'@ function as operator @(?)@
--
-- Since @infixr 0 $@, we can use @?@..@$@ like @?@..@:@ in C-family languages.
--
(?) :: Bool -> p -> p -> p
? :: Bool -> p -> p -> p
(?) p :: Bool
p t :: p
t f :: p
f = if Bool
p then p
t else p
f
infixr 1 ?
